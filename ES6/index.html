Inheritance


					class Human {
						constructor(name){
							this.name = name;
						}

						toString() {
							return "hello " + this.name;
						}
					}

					class Person extends Human{
						constructor(name,age){
						super();
						super.constructor(name);
						this.age = age;
						}
						
						toString(){
							return super.toString() + " I am a person & my age is "+ this.age;
						}
					}

					var me = new Person('kunal',25);
					console.log(me.toString());

Generators

In traditional functions, once the function starts running, it will always run to completion before any other JS code can run

Generators are different kind of functions, which may be paused in the middle, one or many times, and resumed later, allowing other code to run during these paused periods.


					setTimeout(function(){
    					console.log("Hello World");
					},1);

					function foo() {
    					// NOTE: don't ever do crazy long-running loops like this
    					for (var i=0; i<=1E10; i++) {
        					console.log(i);
    					}
					}

					foo();
					
					// 0..1E10
					// "Hello World"

Inside the generator function body, you use the new yield keyword to pause the function from inside itself. Nothing can pause a generator from the outside; it pauses itself when it comes across a yield.

However, once a generator has yield-paused itself, it cannot resume on its own. An external control must be used to restart the generator.


					function *foo(x) {
    					var y = 2 * (yield (x + 1));
    					var z = yield (y / 3);
    					return (x + y + z);
						}

					var it = foo( 5 );

					// note: not sending anything into `next()` here
					console.log( it.next() );       // { value:6, done:false }
					console.log( it.next( 12 ) );   // { value:8, done:false }
					console.log( it.next( 13 ) );   // { value:42, done:true }
